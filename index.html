<script>

    let data = JSON.parse(localStorage.getItem('shift_v5_2_data')) || {

        basic: [],

        sheets: [],

        sheetIdx: 0,

        coolMode: false,

        memo: ""

    };


    function switchTab(id) {

        document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));

        document.querySelectorAll('.mode-btn').forEach(b => b.classList.remove('active'));

        document.getElementById(id).classList.add('active');

        event.currentTarget.classList.add('active');

        render();

    }


    function render() {

        const today = new Date();

        

        // 1. 基礎家事

        const listEl = document.getElementById('taskList');

        const sorted = data.basic.map(t => {

            const last = new Date(t.lastDate);

            const next = new Date(last.getTime() + t.freq * 86400000);

            const diff = Math.floor((today - last) / 86400000);

            return { ...t, next, diff };

        }).sort((a, b) => a.next - b.next);


        listEl.innerHTML = sorted.map(t => `

            <div class="task-item">

                <div>

                    <div class="task-name">${t.name}</div>

                    <div class="task-meta">已過 ${t.diff}d · 預計 ${t.next.getMonth()+1}/${t.next.getDate()}</div>

                </div>

                <button class="primary-btn" onclick="doneBasic('${t.name}')">DONE</button>

            </div>

        `).join('') || '<div style="text-align:center; color:var(--mj-text-sub); font-size:12px; margin-top:20px;">尚無任務，請由上方新增</div>';


        // 2. 床單管理與庫存

        const coolMode = data.coolMode;

        document.getElementById('coolMode').checked = coolMode;

        

        let pool = [];

        const summerSet = data.sheets.filter(s => s.type === 'summer');

        const allSet = data.sheets.filter(s => s.type === 'all');


        if (coolMode) {

            // 加權：2夏季1四季

            const max = Math.max(summerSet.length, allSet.length);

            for(let i=0; i<max; i++) {

                if(summerSet[i % summerSet.length]) pool.push(summerSet[i % summerSet.length]);

                if(summerSet[(i+1) % summerSet.length]) pool.push(summerSet[(i+1) % summerSet.length]);

                if(allSet[i % allSet.length]) pool.push(allSet[i % allSet.length]);

            }

        } else {

            pool = allSet.length > 0 ? allSet : data.sheets;

        }


        const nextS = pool[data.sheetIdx % (pool.length || 1)] || { name: "請新增床包" };

        document.getElementById('nextSheetName').innerText = nextS.name;


        const invEl = document.getElementById('sheetInventory');

        invEl.innerHTML = data.sheets.map((s, idx) => `

            <div class="inventory-item">

                <span onclick="editSheet(${idx})" style="cursor:pointer; color: ${s.type==='summer'? 'var(--mj-terracotta)':'#fff'}; text-decoration:underline;">

                    ${s.name} ${s.type==='summer'?'(夏)':'(通)'}

                </span>

                <button onclick="deleteSheet(${idx})" style="background:transparent; border:none; color:#ff4444; cursor:pointer; font-size:10px;">移除</button>

            </div>

        `).join('') || '<div style="color:var(--mj-text-sub); font-size:11px;">庫存空空如也</div>';


        // 3. 其他

        document.getElementById('memo').value = data.memo;

        document.getElementById('backupCode').value = btoa(unescape(encodeURIComponent(JSON.stringify(data))));

        localStorage.setItem('shift_v5_2_data', JSON.stringify(data));

    }


    // --- 邏輯功能 ---

    function addBasic() {

        const n = document.getElementById('bName').value;

        const v = parseInt(document.getElementById('bVal').value);

        const u = parseInt(document.getElementById('bUnit').value);

        if(n && v) {

            data.basic.push({ name: n, freq: v * u, lastDate: new Date().toISOString() });

            document.getElementById('bName').value = "";

            render();

        }

    }

    function doneBasic(name) {

        data.basic.find(t => t.name === name).lastDate = new Date().toISOString();

        render();

    }

    function addSheet() {

        const n = document.getElementById('sName').value;

        const t = document.getElementById('sType').value;

        if(n) {

            data.sheets.push({ name: n, type: t });

            document.getElementById('sName').value = "";

            render();

        }

    }

    function editSheet(idx) {

        const s = data.sheets[idx];

        const newName = prompt("修改床包名稱：", s.name);

        if (!newName) return;

        const newType = confirm(`設為夏季優先款？\n(確定：是 / 取消：否)`) ? "summer" : "all";

        data.sheets[idx] = { name: newName, type: newType };

        render();

    }

    function deleteSheet(idx) {

        if (confirm(`刪除「${data.sheets[idx].name}」？`)) {

            data.sheets.splice(idx, 1);

            render();

        }

    }

    function toggleCooling() {

        data.coolMode = document.getElementById('coolMode').checked;

        data.sheetIdx = 0;

        render();

    }

    function rotateSheet() { data.sheetIdx++; render(); }

    function saveMemo() { data.memo = document.getElementById('memo').value; localStorage.setItem('shift_v5_2_data', JSON.stringify(data)); }

    function copyBackup() {

        document.getElementById("backupCode").select();

        document.execCommand("copy");

        alert("備份碼已複製到剪貼簿。");

    }

    function importData() {

        const code = prompt("請貼上 Base64 備份碼：");

        if(code) {

            try {

                data = JSON.parse(decodeURIComponent(escape(atob(code))));

                render();

                alert("還原成功！");

            } catch(e) { alert("無效的備份碼"); }

        }

    }


    render(); // 初始渲染看板

</script>
